    def canCompleteCircularArray(self, A: List[int]) -> bool:
        # write your code here
        n = len(A)
        visited = [False] * n
        for i in range(n):
            if visited[A[i]]:
                return False
            visited[A[i]] = True
            next_index = (i + 1) % n
            while visited[A[next_index]]:
                next_index = (next_index + 1) % n
            visited[A[next_index]] = True
        return True
```

from typing import List

class Solution:
    def canCompleteCircularArray(self, A: List[int]) -> bool:
        # Write your code here
        n = len(A)
        visited = [False] * n
        for i in range(n):
            if visited[A[i]]:
                return False
            visited[A[i]] = True
            next_index = (i + 1) % n
            while visited[A[next_index]]:
                next_index = (next_index + 1) % n
            visited[A[next_index]] = True
        return True
