    find_parity([1, 2, 3, 4]) -> {1: 1, 2: 0, 3: 1, 4: 0}

def find_poetry_element(poem, element):
    """
    This function takes a poem and an element as inputs and returns the line number(s) where the specified element is found.
    
    Args:
    poem (str): A string containing the poem.
    element (str): A string representing the element to search for (e.g., "love", "moon", "rhyme").

    Returns:
    list: A list of integers representing the line numbers where the element is found. If the element is not found, return an empty list.
    """
    lines = poem.split("\n")
    occurrences = [i+1 for i, line in enumerate(lines) if element in line]
    return occurrences
```

Sugar is sweet,
And so are you.

def find_poetry_element(poem, element):
    """
    This function takes a poem and an element as inputs and returns the line number(s) where the specified element is found.
    
    Args:
    poem (str): A string containing the poem.
    element (str): A string representing the element to search for (e.g., "love", "moon", "rhyme").

    Returns:
    list: A list of integers representing the line numbers where the element is found. If the element is not found, return an empty list.
    """
    lines = poem.split("\n")
    occurrences = [i+1 for i, line in enumerate(lines) if element in line]
    return occurrences
```

Example usage:

```python
poem = """
Roses are red,
Violets are blue,
Sugar is sweet,
And so are you.
"""

print(find_poetry_element(poem, "red"))  # Output: [1]
print(find_poetry_element(poem, "blue"))  # Output: [2]
print(find_poetry_element(poem, "love"))  # Output: []
```

    if n <= 0:
        return []
    elif n == 1:
        return [0]
    elif n == 2:
        return [0, 1]
